229 inline lfgui::image load_image(std::string path)
230 {
231  Urho3D::String str(path.c_str(),path.size());
232  Urho3D::SharedPtr<Urho3D::Image> i(gui::instance()->_context->GetSubsystem<Urho3D::ResourceCache>()->GetResource<Urho3D::Image>(str));
233 
234  if(!i) // not loaded properly. Urho3D does automatically log this in his error log.
235  return lfgui::image(1,1);
236 
237  image img(i->GetWidth(),i->GetHeight());
238 
239  for(int y=0;y<i->GetHeight();y++)
240  for(int x=0;x<i->GetWidth();x++)
241  {
242  Urho3D::Color c(i->GetPixel(x,y));
243  img.set_pixel(x,y,lfgui::color(c.r_*255,c.g_*255,c.b_*255,c.a_*255));
244  }
245 
246  return img;
247 }
