79 bool clip_line(int& x0,int& y0,int& x1,int& y1,int max_x,int max_y)
80 {
81  // compute outcodes for P0, P1, and whatever point lies outside the clip rectangle
82  int outcode0=compute_out_code(x0,y0,max_x,max_y);
83  int outcode1=compute_out_code(x1,y1,max_x,max_y);
84 
85  while(true)
86  {
87  if(!(outcode0|outcode1)) // Bitwise OR is 0. Trivially accept and get out of loop
88  break;
89  else if (outcode0 & outcode1)
90  return true;
91  else // failed both tests, so calculate the line segment to clip
92  {
93  // from an outside point to an intersection with clip edge
94  int x,y;
95 
96  // At least one endpoint is outside the clip rectangle; pick it.
97  int outcodeOut=outcode0?outcode0:outcode1;
98 
99  // Now find the intersection point;
100  // use formulas y = y0 + slope * (x - x0), x = x0 + (1 / slope) * (y - y0)
101  if(outcodeOut&position::top) // point is above the clip rectangle
102  {
103  x=x0+(x1-x0)*float(max_y-y0)/(y1-y0);
104  y=max_y;
105  }
106  else if(outcodeOut&position::bottom) // point is below the clip rectangle
107  {
108  x=x0+(x1-x0)*float(0-y0)/(y1-y0);
109  y=0;
110  }
111  else if(outcodeOut&position::right) // point is to the right of clip rectangle
112  {
113  y=y0+(y1-y0)*float(max_x-x0)/(x1-x0);
114  x=max_x;
115  }
116  else if(outcodeOut&position::left) // point is to the left of clip rectangle
117  {
118  y=y0+(y1-y0)*float(0-x0)/(x1-x0);
119  x=0;
120  }
121 
122  // Now we move outside point to intersection point to clip
123  // and get ready for next pass.
124  if(outcodeOut==outcode0)
125  {

129  }
