74 bool clip_line(int& x0,int& y0,int& x1,int& y1,int max_x,int max_y)
75 {
76  // compute outcodes for P0, P1, and whatever point lies outside the clip rectangle
77  int outcode0=compute_out_code(x0,y0,max_x,max_y);
78  int outcode1=compute_out_code(x1,y1,max_x,max_y);
79 
80  while(true)
81  {
82  if(!(outcode0|outcode1)) // Bitwise OR is 0. Trivially accept and get out of loop
83  break;
84  else if (outcode0 & outcode1)
85  return true;
86  else // failed both tests, so calculate the line segment to clip
87  {
88  // from an outside point to an intersection with clip edge
89  int x,y;
90 
91  // At least one endpoint is outside the clip rectangle; pick it.
92  int outcodeOut=outcode0?outcode0:outcode1;
93 
94  // Now find the intersection point;
95  // use formulas y = y0 + slope * (x - x0), x = x0 + (1 / slope) * (y - y0)
96  if(outcodeOut&position::top) // point is above the clip rectangle
97  {
98  x=x0+(x1-x0)*float(max_y-y0)/(y1-y0);
99  y=max_y;
100  }
101  else if(outcodeOut&position::bottom) // point is below the clip rectangle
102  {
103  x=x0+(x1-x0)*float(0-y0)/(y1-y0);
104  y=0;
105  }
106  else if(outcodeOut&position::right) // point is to the right of clip rectangle
107  {
108  y=y0+(y1-y0)*float(max_x-x0)/(x1-x0);
109  x=max_x;
110  }
111  else if(outcodeOut&position::left) // point is to the left of clip rectangle
112  {
113  y=y0+(y1-y0)*float(0-x0)/(x1-x0);
114  x=0;
115  }
116 
117  // Now we move outside point to intersection point to clip
118  // and get ready for next pass.
119  if(outcodeOut==outcode0)

124  }
